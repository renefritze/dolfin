#  -l    choose language for ffc-compiler

FFCFLAGS  = -l dolfin

# the build target executable:
TARGET1 = MeshAssemble

all:

	mkdir build; cd build; cmake ..; make;

clean:
	$(RM) $(TARGET1).h
#	$(RM) $(TARGET2).h
	$(RM) -r build CMakeFiles
